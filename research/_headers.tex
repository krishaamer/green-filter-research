\RedeclareSectionCommand[
  beforeskip=0pt,
  afterskip=1em,
  font=\normalfont\bfseries\centering\MakeUppercase
]{section}

% Subsection heading customization
\RedeclareSectionCommand[
  beforeskip=1em,
  afterskip=1em,
  font=\normalfont\bfseries\MakeUppercase
]{subsection}

% Subsubsection heading customization
\RedeclareSectionCommand[
  beforeskip=1em,
  afterskip=1em,
  font=\normalfont\underline,
  afterindent=false
]{subsubsection}

% Adjust numbering in the appendix
\newcommand{\appendixnumbering}{
  \renewcommand{\thesection}{\Alph{section}}
  \renewcommand{\thesubsection}{\thesection.\arabic{subsection}}
  \renewcommand{\thesubsubsection}{\thesubsection.\arabic{subsubsection}}
  \setcounter{section}{0}
  \setcounter{subsection}{0}
  \setcounter{subsubsection}{0}
}
