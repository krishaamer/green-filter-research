\usepackage{sectsty}
\usepackage{titlesec}

% Set chapter (H1) style
\titleformat{\chapter}[display]
  {\centering\bfseries\Large\uppercase} % Centered, bold, large, uppercase
  {CHAPTER \thechapter}{1em}{}

% Set section (H2) style
\titleformat{\section}
  {\centering\bfseries\large} % Centered, bold, title case
  {\thesection}{1em}{}

% Set subsection (H3) style
\titleformat{\subsection}
  {\bfseries\normalsize} % Left-aligned, bold, title case
  {\thesubsection}{1em}{}

% Set subsubsection (H4) style
\titleformat{\subsubsection}
  {\normalsize\underline} % Left-aligned, underlined, title case
  {\thesubsubsection}{1em}{}
